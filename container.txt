# file: bz.Dockerfile

FROM docker.io/library/ubuntu:22.04

ARG PIP_EXTRA_INDEX_URL=http://oa-mirror.mediatek.inc/repository/pypi/simple

# COPY mtk_sources.list /etc/apt/sources.list
# RUN rm /etc/apt/sources.list.d/*

# install system dependencies + Python 3.10 from Ubuntu 22.04 default repo
RUN apt-get update -y \
    && apt-get install -y --no-install-recommends \
        python3.10 \
        python3.10-dev \
        python3.10-venv \
        python3-pip \
        curl \
        wget \
        vim \
        tmux \
        unzip \
        libaio-dev \
        pdsh \
        ccache \
        build-essential \
        cmake \
        zlib1g-dev \
        libbz2-dev \
        liblzma-dev \
        net-tools \
        iperf3 \
        git \
    && rm -rf /var/lib/apt/lists/*

RUN python3.10 -m venv /opt/bz_venv

ENV PATH=/opt/bz_venv/bin:$PATH
ENV PIP_NO_VERIFY_CERTS=1

RUN pip config set global.index-url http://oa-mirror.mediatek.inc/repository/pypi/simple \
    && pip config set global.trusted-host oa-mirror.mediatek.inc \
    && pip config set global.extra-index-url http://oa-mirror.mediatek.inc/repository/pypi/simple \
    && pip config set global.no-cache-dir true

COPY bz_requirements_no_torch.txt /tmp/bz_requirements_no_torch.txt

RUN pip install --upgrade "pip<25" "setuptools<70" wheel \
    && pip install --no-build-isolation -r /tmp/bz_requirements_no_torch.txt

RUN pip config set global.trusted-host download.pytorch.org \
    && pip config set global.trusted-host oa-mirror.mediatek.inc

RUN pip install \
    --index-url https://download.pytorch.org/whl/cu118 \
    --trusted-host download.pytorch.org \
    --trusted-host oa-mirror.mediatek.inc \
    torch==2.3.1+cu118 \
    torchaudio==2.3.1+cu118 \
    torchvision==0.18.1+cu118

WORKDIR /workspace


ENV CUDA_HOME=/opt/conda
ENV PYTHONPATH=/workspace:$PYTHONPATH

ENTRYPOINT []
