# file: bz.Dockerfile

FROM docker.io/library/ubuntu:22.04

# set proxy for building
ARG PIP_EXTRA_INDEX_URL=http://oa-mirror.mediatek.inc/repository/pypi/simple

# upgrade ubuntu source
COPY mtk_sources.list /etc/apt/sources.list
RUN rm -f /etc/apt/sources.list.d/*

# install system dependencies + python/pip
RUN apt-get update -y \
    && apt-get upgrade -y \
    && apt-get install -y --no-install-recommends \
        python3 \
        python3-pip \
        python-is-python3 \
        curl \
        wget \
        vim \
        tmux \
        unzip \
        libaio-dev \
        pdsh \
        ccache \
        build-essential \
        cmake \
        zlib1g-dev \
        libbz2-dev \
        liblzma-dev \
        net-tools \
        iperf3 \
        git \
    && rm -rf /var/lib/apt/lists/*

# set pip.conf & disable pip cache to reduce docker layer size
RUN pip3 config set global.index-url http://oa-mirror.mediatek.inc/repository/pypi/simple \
    && pip3 config set global.trusted-host oa-mirror.mediatek.inc \
    && pip3 config set global.extra-index-url http://oa-mirror.mediatek.inc/repository/pypi/simple \
    && pip3 config set global.no-cache-dir true

# copy requirements into image
COPY bz_requirements.txt /tmp/bz_requirements.txt

# install python dependencies
RUN pip3 install --break-system-packages -r /tmp/bz_requirements.txt

# Set up the expected mount point for source code
WORKDIR /workspace

# Set environment variables
ENV CUDA_HOME=/opt/conda
ENV PYTHONPATH=/workspace:$PYTHONPATH

# remove entrypoint due to LSF bug
ENTRYPOINT []
