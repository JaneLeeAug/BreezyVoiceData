# file: bz.Dockerfile

FROM docker.io/library/ubuntu:22.04

ARG PIP_EXTRA_INDEX_URL=http://oa-mirror.mediatek.inc/repository/pypi/simple

COPY mtk_sources.list /etc/apt/sources.list
RUN rm -f /etc/apt/sources.list.d/*

RUN apt-get update -y \
    && apt-get upgrade -y \
    && apt-get install -y --no-install-recommends \
        python3.10 \
        python3.10-dev \
        python3.10-venv \
        python3-pip \
        curl \
        wget \
        vim \
        tmux \
        unzip \
        libaio-dev \
        pdsh \
        ccache \
        build-essential \
        cmake \
        zlib1g-dev \
        libbz2-dev \
        liblzma-dev \
        net-tools \
        iperf3 \
        git \
    && rm -rf /var/lib/apt/lists/*

RUN python3.10 -m venv /opt/bz_venv

ENV PATH=/opt/bz_venv/bin:$PATH

RUN pip config set global.index-url http://oa-mirror.mediatek.inc/repository/pypi/simple \
    && pip config set global.trusted-host oa-mirror.mediatek.inc \
    && pip config set global.extra-index-url http://oa-mirror.mediatek.inc/repository/pypi/simple \
    && pip config set global.no-cache-dir true

COPY bz_requirements.txt /tmp/bz_requirements.txt

RUN pip install --upgrade pip setuptools wheel \
    && pip install -r /tmp/bz_requirements.txt

WORKDIR /workspace

ENV CUDA_HOME=/opt/conda
ENV PYTHONPATH=/workspace:$PYTHONPATH

ENTRYPOINT []
